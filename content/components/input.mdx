## Usage

```javascript
import { Input } from "react-native-1000Lights";

const ButtonComponent = () => {
  return (
    <Input
      placeholder="9999999999"
      placeholderTextColor="#7d7d7d"
      autoFocus={true}
      maxLength={10}
      keyboardType={"numeric"}
      leftInputIcon={
        <Icon
          name="face"
          size={25}
          // color="white"
          style={{ marginRight: 15 }}
        />
      }
    />
  );
};
```
