## Usage

```javascript
import SelectBox from  "react-native-1000Lights";

const SelectBox = () => {

   const data = 
      {
        placeholder: 'Select',
        label: 'Lorem ipsum ',  
      }

    const options =[
      'option1',
      'option2',
      'option3'
    ]

  return (
    <View>
    <SelectBox data={data} options={options} />
    </View>
  );
};

```
